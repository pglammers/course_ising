\section{Reflection positivity}

\begin{definition}
  \label{def:torus}
    For $n\geq 1$ and $d\geq 1$, we define the torus of circumference $n$ in dimension $d$ as the graph $\T_n^d$ with vertex set
    \(
        \T_n^d := (\Z/n\Z)^d
    \)
    and edge set
    \[
        E(\T_n^d) := \{\{x,y\} : x,y\in \T_n^d, \|x-y\|_1 = 1\}.
    \]
    Write $\langle\blank\rangle_{\T_n^d,\beta}$ for the Ising model on $\T_n^d$ at inverse temperature $\beta$.

    For $m>0$, we define the discrete Gaussian free field with mass $m$ on $\T_n^d$ as the probability measure on $\R^{\T_n^d}$ given by
    \[
        \langle f(\phi)\rangle_{\operatorname{GFF},\T_n^d,\beta,m}
        := \frac{1}{Z_{\operatorname{GFF},\T_n^d,\beta,m}} \int_{\R^{\T_n^d}} f(\phi) e^{-\frac{\beta}{2}(\|\nabla\phi\|_2^2 + m\|\phi\|_2^2)} d\phi,
    \]
    where $\nabla\phi$ is the discrete gradient of $\phi$, defined by
    \[
        \|\nabla\phi\|_2^2 := \sum_{\{x,y\}\in E(\T_n^d)} (\phi_x - \phi_y)^2,
    \]
    and $\|\phi\|_2^2 := \sum_{x\in \T_n^d} \phi_x^2$.
    \end{definition}

\begin{theorem}[Chessboard estimate]
  \label{thm:chessboard}
    For any $n=2^k$, $d\geq 1$, $\beta\geq 0$, $m>0$, and any bounded functions $f_z:\R\to\R$ indexed by $z\in \T_n^d$, we have
    \[
        \langle \prod_{x\in \T_n^d} f_x(\phi_x)\rangle_{\operatorname{GFF},\T_n^d,\beta,m}
        \leq \max_{x\in \T_n^d} \langle \prod_{y\in \T_n^d} f_y(\phi_x)\rangle_{\operatorname{GFF},\T_n^d,\beta,m}.
    \]
\end{theorem}
